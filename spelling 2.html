<body>

    <h1>Spelling Task</h1>

    <p>Rearrange the letters to form the correct word!</p>


    <!-- Scrambled letters -->

    <div class="scrambled-letters" id="scrambled-letters">

        <!-- Letters will be dynamically added here -->

    </div>


    <!-- Drop zones for user to form the word -->

    <div class="user-answer" id="user-answer">

        <!-- Drop zones will be dynamically added here -->

    </div>


    <!-- Feedback section -->

    <div class="feedback" id="feedback"></div>


    <!-- Buttons -->

    <div class="buttons">

        <button class="button" onclick="checkAnswer()">Submit</button>

        <button class="button" onclick="resetTask()">Retry</button>

    </div>


    <script>

        const word = "reduce";

        const scrambled = scrambleWord(word);

        let draggedLetter = null;


        // Initialize the task

        function initializeTask() {

            const scrambledContainer = document.getElementById("scrambled-letters");

            const answerContainer = document.getElementById("user-answer");


            scrambledContainer.innerHTML = "";

            answerContainer.innerHTML = "";


            // Add scrambled letters

            scrambled.split("").forEach(letter => {

                const letterDiv = document.createElement("div");

                letterDiv.className = "letter";

                letterDiv.draggable = true;

                letterDiv.textContent = letter;


                letterDiv.addEventListener("dragstart", () => {

                    draggedLetter = letterDiv;

                    letterDiv.classList.add("dragging");

                });


                letterDiv.addEventListener("dragend", () => {

                    draggedLetter = null;

                    letterDiv.classList.remove("dragging");

                });


                scrambledContainer.appendChild(letterDiv);

            });


            // Add drop zones

            for (let i = 0; i < word.length; i++) {

                const dropZone = document.createElement("div");

                dropZone.className = "drop-zone";


                dropZone.addEventListener("dragover", event => {

                    event.preventDefault();

                });


                dropZone.addEventListener("drop", () => {

                    if (!dropZone.classList.contains("occupied") && draggedLetter) {

                        dropZone.textContent = draggedLetter.textContent;

                        dropZone.classList.add("occupied");

                        draggedLetter.remove();

                    }

                });


                answerContainer.appendChild(dropZone);

            }

        }


        // Scramble the word

        function scrambleWord(word) {

            const letters = word.split("");

            for (let i = letters.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [letters[i], letters[j]] = [letters[j], letters[i]];

            }

            return letters.join("");

        }


        // Check if the user's answer is correct

        function checkAnswer() {

            const answerContainer = document.getElementById("user-answer");

            const feedback = document.getElementById("feedback");


            const userAnswer = Array.from(answerContainer.children)

                .map(zone => zone.textContent)

                .join("");


            if (userAnswer === word) {

                feedback.textContent = "Correct! Well done!";

                feedback.className = "feedback correct";

            } else {

                feedback.textContent = "Incorrect. Try again!";

                feedback.className = "feedback incorrect";

            }

        }


        // Reset the task

        function resetTask() {

            const feedback = document.getElementById("feedback");

            feedback.textContent = "";

            feedback.className = "feedback";

            initializeTask();

        }


        // Initialize on page load

        initializeTask();

    </script>
